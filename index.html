การเชื่อมต่อชุดข้อมูลเข้า Google Sheets (แผ่นงาน: AuditData)

จับคู่และปรับฟอร์แมตอัตโนมัติ สวยงาม เข้าใจง่าย ผ่าน Google Apps Script บนคลาวด์

1. ตั้งค่า Webhook URL (Google Web App)

2. ซอร์สโค้ด Google Apps Script สำหรับแท็บแผ่นงาน `AuditData` และการจำการตั้งค่าอัตโนมัติ (`CONFIG`)

นำรหัสโค้ดที่ได้รับการพัฒนาให้จัดรูปแบบสีสันและแสดงผลภาษาไทยอย่างเป็นระบบชุดนี้ ไปติดตั้งในเมนู **Extensions > Apps Script** เพื่อสร้างตารางแดชบอร์ดที่สมบูรณ์แบบบน Spreadsheet ของคุณ:

/* ... existing code ... */
// เพิ่มฟังก์ชัน doGet เพื่อให้เครื่องอื่นดึงข้อมูลไปใช้ได้
function doGet(e) {
  var action = e.parameter.action;
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  
  if (action === "fetch") {
    var sheet = ss.getSheetByName("AuditData");
    var data = sheet.getDataRange().getValues();
    // แปลงข้อมูลเป็น JSON
    var jsonData = JSON.stringify(data);
    return ContentService.createTextOutput(jsonData).setMimeType(ContentService.MimeType.JSON);
  }

  // ดึงค่า config ปกติ
  var sheetConfig = ss.getSheetByName("CONFIG");
  var url = sheetConfig ? sheetConfig.getRange("A2").getValue() : "";
  return ContentService.createTextOutput(JSON.stringify({result: "success", webhook_url: url}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    // 1. ตรวจสอบว่าเป็นการบันทึกการตั้งค่า Webhook จากอุปกรณ์หลักหรือไม่
    if (params.action === "save_config") {
      var configSheet = ss.getSheetByName("CONFIG");
      if (!configSheet) {
        configSheet = ss.insertSheet("CONFIG");
        configSheet.appendRow(["Webhook URL", "Last Updated"]);
      }
      configSheet.getRange("A2").setValue(params.webhook_url);
      configSheet.getRange("B2").setValue(new Date());
      return ContentService.createTextOutput(JSON.stringify({result: "success"}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    // 2. การทำงานบันทึกผลแบบประเมินหลัก
    var sheet = ss.getSheetByName("AuditData");
    var isNew = false;
    if (!sheet) {
      sheet = ss.insertSheet("AuditData");
      isNew = true;
    }
    
    // ตั้งชื่อหัวคอลัมน์ภาษาไทยระบุตัวข้อชัดเจนเพื่อให้อ่านง่ายสไตล์ Dashboard
    var headers = [
      "วันเวลาบันทึก (Timestamp)", 
      "วันที่ประเมิน (Audit Date)", 
      "1. แจ้ง ICN ทันที", 
      "2. แจ้งบุคลากรทราบ", 
      "3. แยกผู้ป่วย/โซน", 
      "4. ติดป้าย Contact precautions", 
      "5. ไม่วางของบนเตียง", 
      "6. แยกอุปกรณ์ของใช้", 
      "7. เตรียม Alcohol gel", 
      "8. ล้างมือด้วยน้ำยา 4% CHG", 
      "9. สวม PPE (ถุงมือ/เสื้อคลุม/Mask)", 
      "10. แยกผ้าและขยะติดเชื้อ", 
      "11. ทำความสะอาดสิ่งแวดล้อม", 
      "12. เคลื่อนย้ายแจ้งหน่วยงาน", 
      "13. พนักงานเปลสวม PPE", 
      "14. จำกัดบุคลากร/ผู้เฝ้าไข้", 
      "15. แนะนำผู้ป่วยและญาติ", 
      "16. ทำการเพาะเชื้อเพื่อยุติ", 
      "ปฏิบัติ (YES)", 
      "ไม่ปฏิบัติ (NO)", 
      "อัตราปฏิบัติถูกต้อง (Compliance %)", 
      "บันทึกเพิ่มเติมภาพรวม (generalNotes)"
    ];
    
    if (isNew || sheet.getLastRow() === 0) {
      sheet.appendRow(headers);
      formatHeaderAndApplyRules(sheet);
    }
    
    // แมปสถานะเป็นภาษาไทยเพื่อง่ายต่อการสรุปผลและดูสถิติ
    function mapStatus(val) {
      if (val === "yes") return "ปฏิบัติ";
      if (val === "no") return "ไม่ปฏิบัติ";
      return "ไม่เกี่ยวข้อง (N/A)";
    }
    
    var rowData = [
      new Date(),
      params.audit_date || "",
      mapStatus(params.check1),
      mapStatus(params.check2),
      mapStatus(params.check3),
      mapStatus(params.check4),
      mapStatus(params.check5),
      mapStatus(params.check6),
      mapStatus(params.check7),
      mapStatus(params.check8),
      mapStatus(params.check9),
      mapStatus(params.check10),
      mapStatus(params.check11),
      mapStatus(params.check12),
      mapStatus(params.check13),
      mapStatus(params.check14),
      mapStatus(params.check15),
      mapStatus(params.check16),
      params.score_yes || 0,
      params.score_no || 0,
      parseFloat(params.compliance_percent || 0) / 100, // ส่งเป็นค่าทศนิยมสำหรับจัดฟอร์แมต % ในชีต
      params.generalNotes || ""
    ];
    
    sheet.appendRow(rowData);
    
    // ตั้งค่าฟอร์แมตตัวเลขคอลัมน์ % ให้เป็นสากลใน Google Sheet
    var lastRow = sheet.getLastRow();
    sheet.getRange(lastRow, 21).setNumberFormat("0.0%");
    sheet.getRange(lastRow, 1, 1, sheet.getLastColumn()).setVerticalAlignment("middle");
    
    return ContentService.createTextOutput(JSON.stringify({result: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({result: "error", error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// ฟังก์ชันตกแต่งชีตและใช้กฎจัดกลุ่มเฉดสีอัตโนมัติ (Vibrant Cyan & Smart Status colors)
function formatHeaderAndApplyRules(sheet) {
  var headerRange = sheet.getRange(1, 1, 1, 22);
  headerRange.setBackground("#0891b2") // โทนสีฟ้าครามสดใส (Cyan 600) ให้ความรู้สึกเป็นแอปการแพทย์สมัยใหม่
             .setFontColor("#ffffff")
             .setFontWeight("bold")
             .setHorizontalAlignment("center")
             .setVerticalAlignment("middle")
             .setWrap(true);
             
  sheet.setRowHeight(1, 42);
  sheet.setFrozenRows(1);
  sheet.setFrozenColumns(2); // ตรึงคอลัมน์เวลาและวันที่ประเมิน
  
  // ปรับความกว้างของคอลัมน์ให้เหมาะกับขนาดคำอธิบายภาษาไทย
  sheet.autoResizeColumns(1, 22);
  
  var rules = [];
  
  // กฎ 1: คำว่า "ปฏิบัติ" -> แสดงแถบสีเขียวอ่อนมีชีวิตชีวา
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenTextEqualTo("ปฏิบัติ")
    .setBackground("#d1fae5")
    .setFontColor("#065f46")
    .setRanges([sheet.getRange("C2:R1000")])
    .build());
    
  // กฎ 2: คำว่า "ไม่ปฏิบัติ" -> แสดงแถบสีแดงอ่อนเพื่อส่งสัญญาณเฝ้าระวัง
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenTextEqualTo("ไม่ปฏิบัติ")
    .setBackground("#ffe4e6")
    .setFontColor("#9f1239")
    .setRanges([sheet.getRange("C2:R1000")])
    .build());
    
  // กฎ 3: คำว่า "ไม่เกี่ยวข้อง (N/A)" -> ปรับเป็นเฉดสีเทาสะอาดตา
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenTextEqualTo("ไม่เกี่ยวข้อง (N/A)")
    .setBackground("#f1f5f9")
    .setFontColor("#64748b")
    .setRanges([sheet.getRange("C2:R1000")])
    .build());
    
  // กฎ 4: คะแนนรวมอัตราการปฏิบัติต้องผ่านเกณฑ์ >= 80% เป็นสีเขียวเข้มเด่น
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenNumberGreaterThanOrEqualTo(0.8)
    .setBackground("#d1fae5")
    .setFontColor("#065f46")
    .setRanges([sheet.getRange("U2:U1000")])
    .build());
    
  // กฎ 5: คะแนนรวมต่ำกว่าเกณฑ์ < 50% เป็นสีแดงเน้นย้ำเพื่อให้ได้รับการแก้ด่วน
  rules.push(SpreadsheetApp.newConditionalFormatRule()
    .whenNumberLessThan(0.5)
    .setBackground("#ffe4e6")
    .setFontColor("#9f1239")
    .setRanges([sheet.getRange("U2:U1000")])
    .build());
    
  sheet.setConditionalFormatRules(rules);
}
